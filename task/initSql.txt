nginx+php上传大文件配置
//nginx
'client_max_body_size' => '500M',  #上传文件大小限制
//php
'upload_max_filesize' => '500M',  #允许上传文件大小的最大值,需在php.ini中配置
'post_max_size' => '500M', #通过表单POST给PHP的所能接收的最大值，包括表单里的所有值,需在php.ini中配置
'max_execution_time' => '600', #每个PHP页面运行的最大时间值(秒),仅对php页面有作用,可在代码中修改
'max_input_time' => '600', #每个PHP页面接收数据所需的最大时间,仅对php页面有作用,可在代码中修改
'memory_limit' => '600M', #每个PHP页面所吃掉的最大内存,仅对php页面有作用,可在代码中修改
